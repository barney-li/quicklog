//------------------------------------------------------------------------
// 简称: myCrossOver
// 名称: CrossOver
// 类别: 用户函数
// 类型: 用户函数
// 输出: 数值型
//------------------------------------------------------------------------

Params
	NumericSeries price;
	NumericSeries base;
	Numeric underDistance;
Vars
	Bool crossed;
	Numeric index;
Begin
	
	If(price>base)
	{
		crossed = True;
		For index=2 To underDistance
		{
			If(price[index]<base[index])
			{
				crossed = True;
			}
			Else
			{
				crossed = False;
				break;
			}
		}
	}
	Return crossed; 
End
