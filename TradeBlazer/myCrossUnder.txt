//------------------------------------------------------------------------
// 简称: myCrossUnder
// 名称: CrossUnder
// 类别: 用户函数
// 类型: 用户函数
// 输出: 布尔型
//------------------------------------------------------------------------

Params
	NumericSeries price;
	NumericSeries base;
	Numeric overDistance;
Vars
	Bool crossed;
	Numeric index;
Begin
	
	If(price<base)
	{
		crossed = True;
		For index=2 To overDistance
		{
			If(price[index]>base[index])
			{
				crossed = True;
			}
			Else
			{
				crossed = False;
				break;
			}
		}
	}
	Return crossed; 
End
