// 简称: mySigmaSignedVol
// 名称: SigmaSignedVol
// 类别: 用户函数
// 类型: 用户函数
// 输出: 数值型
//------------------------------------------------------------------------

Params
	Numeric barIndex(0);
Vars
	Numeric index(0);
	Numeric sigmaVol(0);
Begin
	For index=0 to barIndex
	{
		sigmaVol = sigmaVol+IIF(openint[index]>openint[index+1], 1, -1)*Vol[index];
	}
	Return sigmaVol;
End
