// 类别: 用户函数
// 类型: 用户函数
// 输出: 数值型
//------------------------------------------------------------------------

Params
	Numeric getOutTime(14.55);
	Numeric slide(1);
	Numeric volume(1);
Vars
	Numeric closePrice(0);
Begin
	if(time > getOutTime/100)
	{
		if(MarketPosition == 1) 
		{
			closePrice = Close - slide;
			sell(volume, closePrice);
		}
		if(MarketPosition == -1) 
		{
			closePrice = close + slide;
			BuyToCover(volume, closePrice);
		}
	}
	Return closePrice;
End
