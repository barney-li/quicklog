Params
	NumericSeries price;
	Numeric length;
Vars
	Numeric sum(0);
	Numeric denoisedPrice(0);
Begin
	denoisedPrice = price;
	sum = SummationFC(price, length)-price;
	//While(denoisedPrice>=sum)
	//{
	//	denoisedPrice = denoisedPrice/2;
	//}
	denoisedPrice = sum;
	Return denoisedPrice;
End
